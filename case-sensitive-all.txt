SELECT ID, 'content' as type from wp_posts
  WHERE post_content
  LIKE BINARY  '([[:blank:][:punct:]]|^)Alligator([[:blank:][:punct:]]|$)'
  OR post_title
  LIKE BINARY  '([[:blank:][:punct:]]|^)Alligator([[:blank:][:punct:]]|$)'
  AND post_type IN (
    'interview',
    'interactive',
    'timeline',
    'post'
  )
UNION
SELECT post_id, 'content' as type from wp_postmeta
  WHERE meta_value
  LIKE BINARY '([[:blank:][:punct:]]|^)Alligator([[:blank:][:punct:]]|$)'
  AND (
    meta_key IN (
    'transcript_raw',
    'introduction',
    'description_raw',
    'supporting_content_raw'
    )
    OR meta_key LIKE 'events_%_content'
    OR meta_key LIKE 'events_%_title'
  )
UNION
SELECT term_id, 'term' as type from wp_terms
  WHERE name
  LIKE BINARY  '([[:blank:][:punct:]]|^)Alligator([[:blank:][:punct:]]|$)'
UNION
SELECT term_id, 'term' as type from wp_termmeta
  WHERE meta_value
  LIKE BINARY  '([[:blank:][:punct:]]|^)Alligator([[:blank:][:punct:]]|$)'
  AND meta_key IN ('collection_description')
